51 size_t i = 0, ans = 0;
52 while (i < n) {
53 ans = i;
54 size_t j = i + 1, k = i;
55 while (j < (n + n) && s[j % n] >= s[k % n]) {
56 if (s[k % n] < s[j % n]) {
57 k = i;
58 } else {
59 k++;
60 }
61 j++;
62 }

64 i += j - k;
65 }
