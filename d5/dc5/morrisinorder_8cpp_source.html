16void insert(Btree **root, int d) {

18 nn->data = d;

22 *root = nn;

25 queue<Btree *> q;

27 q.push(*root);

29 Btree *node = q.front();
30 // Removing parent node from queue
31 q.pop();

33 // Adding left child of removed node to queue
34 q.push(node->left);

36 // Adding new node if no left child is present
37 node->left = nn;

42 q.push(node->right);
