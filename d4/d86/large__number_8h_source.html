267 temp = static_cast<uint64_t>((*this)[i]) * n;
268 temp += carry;
269 if (temp < 10) {
270 carry = 0;
271 } else {
272 carry = temp / 10;
273 temp = temp % 10;
274 }
275 (*this)[i] = temp;
276 }
277
278 while (carry != 0) {
